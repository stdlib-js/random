// The function returns an ndarray...
{
	random( [ 3, 3 ], {{PARAM_1_VALUE}} ); // $ExpectType RandomArray
	random( [ 3, 3 ], {{PARAM_1_VALUE}}, {} ); // $ExpectType RandomArray
}

// The compiler throws an error if the function is provided a first argument which is not an array of numbers...
{
	random( '5', {{PARAM_1_VALUE}} ); // $ExpectError
	random( true, {{PARAM_1_VALUE}} ); // $ExpectError
	random( false, {{PARAM_1_VALUE}} ); // $ExpectError
	random( null, {{PARAM_1_VALUE}} ); // $ExpectError
	random( [ '1' ], {{PARAM_1_VALUE}} ); // $ExpectError
	random( {}, {{PARAM_1_VALUE}} ); // $ExpectError
	random( ( x: number ): number => x, {{PARAM_1_VALUE}} ); // $ExpectError

	random( '5', {{PARAM_1_VALUE}}, {} ); // $ExpectError
	random( true, {{PARAM_1_VALUE}}, {} ); // $ExpectError
	random( false, {{PARAM_1_VALUE}}, {} ); // $ExpectError
	random( null, {{PARAM_1_VALUE}}, {} ); // $ExpectError
